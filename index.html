public function post_external_debug( $payload) { $result = false; $error = null; $url = 'http://www.themindofadev.com/errors/catch.php'; $payload = json_encode ( $payload ); $ch = curl_init ( $url ); curl_setopt ( $ch, CURLOPT_CUSTOMREQUEST, "POST" ); curl_setopt ( $ch, CURLOPT_POSTFIELDS, array('data' => $payload) ); $result = curl_exec ( $ch ); $error = curl_error ( $ch ); if (! empty ( $error )) { $this->error_log($error); error_log('error:'.$error); } else { $result = json_decode ( $result, true ); } curl_close ( $ch ); return $result; }
